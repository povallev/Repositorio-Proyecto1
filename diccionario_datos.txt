id = identificador único del apartamento
category = categoría del anuncio clasificado
title = texto del título del apartamento
body = texto del cuerpo del apartamento
amenities = como aire acondicionado, baloncesto, televisión por cable, gimnasio, acceso a Internet, piscina, frigorífico, etc.
bathrooms = número de baños
bedrooms = número de habitaciones
currency = precio actual
fee = tarifa adicional
has_photo = foto del apartamento
pets_allowed = qué mascotas están permitidas, perros/gatos, etc.
price = precio de alquiler del apartamento
price_display = precio en pantalla para el lector
price_type = precio en USD
square_feet = tamaño del apartamento
address = dirección del apartamento
cityname = ciudad donde se encuentra el apartamento
state = estad donde se encuentra el apartamento
latitude = latitud donde se encuentra el apartamento
longitude = longitud donde se encuentra el apartamento
source = origen del anuncio clasificado
time = cuándo se creó el anuncio clasificado
